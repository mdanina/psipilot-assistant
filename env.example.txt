# PsiPilot Assistant - Environment Variables Example
# Скопируйте этот файл в .env.local и заполните своими значениями

# Supabase Configuration (Self-hosted)
# URL вашего Supabase API (например, http://localhost:8000 или https://your-domain.com)
VITE_SUPABASE_URL=http://localhost:8000

# Ваш Supabase anon/public ключ
# Найти можно в Supabase Dashboard > Settings > API > Project API keys > anon public
VITE_SUPABASE_ANON_KEY=your-anon-key-here

# Опционально: Service Role Key (только для server-side операций, никогда не используйте в клиенте!)
# VITE_SUPABASE_SERVICE_ROLE_KEY=your-service-role-key-here
