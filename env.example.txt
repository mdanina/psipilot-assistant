# PsiPilot Assistant - Environment Variables Example
# Скопируйте этот файл в .env.local и заполните своими значениями

# Supabase Configuration (Self-hosted)
# URL вашего Supabase API (например, http://localhost:8000 или https://your-domain.com)
VITE_SUPABASE_URL=http://localhost:8000

# Ваш Supabase anon/public ключ
# Найти можно в Supabase Dashboard > Settings > API > Project API keys > anon public
VITE_SUPABASE_ANON_KEY=your-anon-key-here

# Service Role Key (для создания пользователей через скрипт)
# ⚠️ ВАЖНО: Никогда не используйте в клиентском коде! Только для server-side операций!
# Найти можно в Supabase Dashboard > Settings > API > Project API keys > service_role
# Используется скриптом scripts/create-user.js для создания тестовых пользователей
SUPABASE_SERVICE_ROLE_KEY=your-service-role-key-here

# ============================================
# SECURITY CONFIGURATION (HIPAA/GDPR)
# ============================================

# ВАЖНО: Ключ шифрования теперь настраивается на BACKEND!
# См. backend/transcription-service/.env.example -> ENCRYPTION_KEY
# Это обеспечивает безопасность - ключ никогда не передается в браузер

# Session Timeout (в миллисекундах, по умолчанию 15 минут)
# VITE_SESSION_TIMEOUT=900000

# TRANSCRIPTION SERVICE CONFIGURATION
# ============================================
